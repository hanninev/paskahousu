33      /**
34       * Metodi kertoo, mikä pelaajan kädessä olevista korteista on arvoltaan
35       * pienin.
36       *
37       * @return pienin kortti
38       */
39      public Kortti pieninKortti() {
40          Kortti pienin = kasi.getKortit().get(0);
41          for (Kortti kortti : kasi.getKortit()) {
42              if (kortti.getArvo() <= pienin.getArvo()) {
43                  pienin = kortti;
44              }
45          }
46          return pienin;
47      }
48  
49      /**
50       * Metodilla voidaan lisätä kyseisen käyttäjän käteen haluttu kortti.
51       *
52       * @param kortti Käteen lisättävä kortti
53       */
54      public void lisaaKateen(Kortti kortti) {
55          kasi.lisaa(kortti);
56      }
57  
58      /**
59       * Metodilla voidaan ottaa kyseisen pelaajan kädestä haluttu kortti.
60       *
61       * @param kortti Kädestä otettava kortti
62       *
63       * @return Kädestä otettu kortti
64       */
65      public Kortti otaKadesta(Kortti kortti) throws Exception {
66          return kasi.ota(kortti);
67      }

70      public int hashCode() {

85              return false;

92      }
