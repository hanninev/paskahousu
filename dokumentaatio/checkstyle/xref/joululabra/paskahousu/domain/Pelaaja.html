31      /**
32       * Metodi kertoo, mikä pelaajan kädessä olevista korteista on arvoltaan
33       * pienin.
34       *
35       * @return pienin kortti
36       */
37      public Kortti pieninKortti() {
38          Kortti pienin = kasi.getKortit().get(0);
39          for (Kortti kortti : kasi.getKortit()) {
40              if (kortti.getArvo() <= pienin.getArvo()) {
41                  pienin = kortti;
42              }
43          }
44          return pienin;
45      }

47      /**
48       * Metodilla voidaan lisätä kyseisen käyttäjän käteen haluttu kortti.
49       *
50       * @param kortti Käteen lisättävä kortti
51       */
52      public void lisaaKateen(Kortti kortti) {
53          kasi.lisaa(kortti);
54      }
55  
56      /**
57       * Metodilla voidaan ottaa kyseisen pelaajan kädestä haluttu kortti.
58       *
59       * @param kortti Kädestä otettava kortti
60       *
61       * @return Kädestä otettu kortti
62       */
63      public Kortti otaKadesta(Kortti kortti) throws Exception {
64          return kasi.ota(kortti);
65      }
