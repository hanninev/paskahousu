11  public class Paaohjelma {
12  
13      public static void main(String[] args) throws Exception {
14          // väliaikainen käyttöliittymä, jolla voi testata toiminnallisuutta
15  
16          Scanner lukija = new Scanner(System.in);
17          Peli peli = new Peli();
18          Siirtojenkasittelija sk = new Siirtojenkasittelija();
19  
20          Pelaaja ville = new Pelaaja();
21          ville.setNimi("Ville");
22          peli.lisaaPelaaja(ville);
23  
24          Pelaaja jani = new Pelaaja();
25          jani.setNimi("Jani");
26          peli.lisaaPelaaja(jani);
27  
28          peli.alusta();
29  
30          while (peli.peliJatkuu()) {
31              for (Pelaaja pelaaja : peli.getPelaajat()) {
32  
33                  Vuoro vuoro = peli.lisaaVuoro(pelaaja);
34                  if (Saannot.joutuuAinaNostamaanPinon(peli.getPino())) {
35                      sk.nostaPino(peli.getPino(), vuoro);
36                      break;
37                  }
38                  while (vuoro.jatkuu()) {
39                      System.out.println("Pakkaa jäljellä: " + peli.getPakka().korttienMaara());
40                      System.out.println("Kortteja pinossa: " + peli.getPino().korttienMaara());
41                      System.out.println("Pinon ylin kortti: " + peli.getPino().viimeisinKortti());
42                      System.out.println("Kortit kädessä: " + pelaaja.getKasi().getKortit());
43                      System.out.println(pelaaja.getNimi() + ", valitse toiminto.");
44                      System.out.println("Komennot:");
45                      System.out.println("a = lisää kortti pinoon");
46                      System.out.println("b = kokeile onnea pakasta");
47                      System.out.println("c = nosta pino");
48                      System.out.println("x = olen valmis, seuraavan vuoro");
49                      String komento = lukija.nextLine();
50  
51                      if (komento.equals("a")) {
52                          System.out.println("valitse kortti:");
53                          int maa = Integer.parseInt(lukija.nextLine());
54                          int arvo = Integer.parseInt(lukija.nextLine());
55                          sk.siirraKorttiPinoon(new Kortti(maa, arvo), peli.getPakka(), peli.getKaatuneetKortit(), peli.getPino(), vuoro);
56  
57                      } else if (komento.equals("b")) {
58                          sk.kokeileOnnea(peli.getPakka(), peli.getPino(), peli.getKaatuneetKortit(), vuoro);
59                          break;
60  
61                      } else if (komento.equals("c")) {
62                          sk.nostaPino(peli.getPino(), vuoro);
63                          break;
64  
65                      } else if (komento.equals("x")) {
66                          vuoro.setJatkuu(false);
67                          sk.taydennaKasi(peli.getPakka(), vuoro);
68                          break;
69                      }
70                  }
71              }
72          }
73      }
74  }
