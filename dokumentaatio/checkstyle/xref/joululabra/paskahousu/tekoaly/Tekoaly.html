1   package joululabra.paskahousu.tekoaly;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import joululabra.paskahousu.domain.Kortti;
6   import joululabra.paskahousu.sovelluslogiikka.Saannot;
7   import joululabra.paskahousu.sovelluslogiikka.Siirtojenkasittelija;
8   
9   /**
10   * Luokka tarjoaa tekoälyyn liittyviä metodeita.
11   */
12  public class Tekoaly {
13  
14      private Siirtojenkasittelija sk;
15  
16      /**
17       * Konstruktori tekee uuden olioilmentymän luokasta Tekoaly.
18       *
19       * @param sk Tekoälyyn liitettävä siirtojenkäsittelijä.
20       */
21      public Tekoaly(Siirtojenkasittelija sk) {
22          this.sk = sk;
23      }
24  
25      /**
26       * Metodi valitsee toiminnon.
27       *
28       * @throws Exception Siirtoa ei voida tehdä, jos siirrettävää korttia ei ole
29       * kädessä.
30       */
31      public void valitseToiminto() throws Exception {
32          try {
33              if (Saannot.pakkoNostaaPino(sk.getPino())) {
34                  sk.nostaPino();
35                  return;
36              }
37  
38              if (!kaykoMikaanKortti()) {
39                  sk.kokeileOnnea();
40              } else if (kaykoMikaanKortti()) {
41                  while (kaykoMikaanKortti()) {
42                      laitaKortti();
43                  }
44              } else {
45                  sk.nostaPino();
46              }
47          } catch (Exception e) {
48              System.out.println(e);
49          }
50      }
51  
52      /**
53       * Metodi laittaa pinoon kortteja.
54       *
55       * @throws Exception Siirtoa ei voida tehdä, jos siirrettävää korttia ei ole
56       * kädessä.
57       */
58      public void laitaKortti() throws Exception {
59          List<Kortti> kortitKadessa = new ArrayList<>(sk.nykyinenVuoro().getPelaaja().getKasi().getKortit());
60  
61          for (Kortti kortti : kortitKadessa) {
62              if (!sk.nykyinenVuoro().getPelaaja().getKasi().onTyhja()) {
63                  if (Saannot.korttiSopii(sk.getPino(), sk.getPakka(), sk.nykyinenVuoro(), kortti)) {
64                      sk.siirraKorttiPinoon(kortti);
65                  }
66              }
67          }
68      }
69  
70      /**
71       * Metodi kertoo, käykö mikään kädessä oleva kortti pinoon.
72       * 
73       * @return boolean
74       */
75      public boolean kaykoMikaanKortti() {
76          boolean onnistui = false;
77  
78          for (Kortti kortti : sk.nykyinenVuoro().getPelaaja().getKasi().getKortit()) {
79              if (!sk.nykyinenVuoro().getPelaaja().getKasi().onTyhja()) {
80                  if (Saannot.korttiSopii(sk.getPino(), sk.getPakka(), sk.nykyinenVuoro(), kortti)) {
81                      onnistui = true;
82                  }
83              }
84          }
85          return onnistui;
86      }
87  }
