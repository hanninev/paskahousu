1   package joululabra.paskahousu.sovelluslogiikka;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import joululabra.paskahousu.domain.Pelaaja;
6   
7   public class Peli {
8   
9       private Siirtojenkasittelija sk;
10      private Tekoaly tekoaly;
11      private List<Pelaaja> pelaajat;
12  
13      public Peli() {
14          sk = new Siirtojenkasittelija();
15          pelaajat = new ArrayList<>();
16          tekoaly = new Tekoaly(sk);
17      }
18  
19      public Siirtojenkasittelija getSk() {
20          return sk;
21      }
22  
23      public List<Pelaaja> getPelaajat() {
24          return pelaajat;
25      }
26  
27      public Tekoaly getTekoaly() {
28          return tekoaly;
29      }
30  
31      public void lisaaPelaaja(String nimi) {
32          this.pelaajat.add(new Pelaaja(nimi));
33      }
34  
35      public void jaaKortit() throws Exception {
36          for (Pelaaja pelaaja : pelaajat) {
37              for (int i = 0; i < 5; i++) {
38                  pelaaja.lisaaKateen(sk.getPakka().otaEnsimmainenKortti());
39              }
40          }
41      }
42  
43      public void sekoitaPakka() {
44          sk.getPakka().sekoitaKortit();
45      }
46  
47      /**
48       * Metodi kertoo, että peli loppuu, jos vain yhdellä pelaajalla on kädessään kortteja.
49       *
50       * @return boolean
51       */
52      public boolean peliJatkuu() {
53          int pelissaMukana = 0;
54          for (Pelaaja pelaaja : pelaajat) {
55              if (!pelaaja.getKasi().onTyhja()) {
56                  pelissaMukana++;
57              }
58          }
59          return pelissaMukana != 1;
60      }
61  }
