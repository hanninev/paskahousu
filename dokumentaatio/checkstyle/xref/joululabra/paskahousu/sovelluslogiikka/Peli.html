1   package joululabra.paskahousu.sovelluslogiikka;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import joululabra.paskahousu.domain.Pelaaja;
6   import joululabra.paskahousu.tekoaly.Tekoaly;
7   
8   /**
9    * Luokka tarjoaa pelin alustamiseen tarvittavia metodeja.
10   */
11  public class Peli {
12  
13      private Siirtojenkasittelija sk;
14      private Tekoaly tekoaly;
15      private List<Pelaaja> pelaajat;
16  
17      /**
18       * Metodi tekee uuden olioilmentymän luokasta Peli.
19       *
20       */
21      public Peli() {
22          sk = new Siirtojenkasittelija(this);
23          pelaajat = new ArrayList<>();
24          tekoaly = new Tekoaly(sk);
25      }
26  
27      public Siirtojenkasittelija getSk() {
28          return sk;
29      }
30  
31      public List<Pelaaja> getPelaajat() {
32          return pelaajat;
33      }
34  
35      public Tekoaly getTekoaly() {
36          return tekoaly;
37      }
38  
39      /**
40       * Metodi lisää peliin uuden pelaajan.
41       *
42       * @param nimi Pelaajan nimi
43       *
44       */
45      public void lisaaPelaaja(String nimi) {
46          this.pelaajat.add(new Pelaaja(nimi));
47      }
48  
49      /**
50       * Metodi jakaa kaikille pelaajille viisi korttia pakasta.
51       *
52       */
53      public void jaaKortit() {
54          for (Pelaaja pelaaja : pelaajat) {
55              for (int i = 0; i < 5; i++) {
56                  try {
57                      pelaaja.lisaaKateen(sk.getPakka().otaEnsimmainenKortti());
58                  } catch (Exception e) {
59                      System.out.println(e.getMessage());
60                  }
61              }
62          }
63      }
64  
65      /**
66       * Metodi sekoittaa pakan.
67       *
68       */
69      public void sekoitaPakka() {
70          sk.getPakka().sekoitaKortit();
71      }
72  
73      /**
74       * Metodi kertoo, että peli loppuu, jos vain yhdellä pelaajalla on kädessään
75       * kortteja.
76       *
77       * @return boolean
78       */
79      public boolean peliJatkuu() {
80          int pelissaMukana = 0;
81          for (Pelaaja pelaaja : pelaajat) {
82              if (!pelaaja.getKasi().onTyhja()) {
83                  pelissaMukana++;
84              }
85          }
86          return pelissaMukana != 1;
87      }
88  
89      /**
90       * Metodi asettaa seuraavan pelaajan vuoroon.
91       *
92       */
93      public void asetaSeuraavaPelaajaVuoroon() {
94          for (int i = 0; i < pelaajat.size(); i++) {
95              if (pelaajat.get(i).equals(sk.nykyinenVuoro().getPelaaja())) {
96                  sk.nykyinenVuoro().setJatkuu(false);
97                  if (i == pelaajat.size() - 1) {
98                      sk.lisaaVuoro(pelaajat.get(0));
99                      break;
100                 } else {
101                     sk.lisaaVuoro(pelaajat.get(i + 1));
102                     break;
103                 }
104             }
105         }
106     }
107 }
