1   package joululabra.paskahousu.sovelluslogiikka;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import joululabra.paskahousu.domain.Pelaaja;
6   import joululabra.paskahousu.tekoaly.Tekoaly;
7   
8   public class Peli {
9   
10      private Siirtojenkasittelija sk;
11      private Tekoaly tekoaly;
12      private List<Pelaaja> pelaajat;
13  
14      public Peli() {
15          sk = new Siirtojenkasittelija(this);
16          pelaajat = new ArrayList<>();
17          tekoaly = new Tekoaly(sk);
18      }
19  
20      public Siirtojenkasittelija getSk() {
21          return sk;
22      }
23  
24      public List<Pelaaja> getPelaajat() {
25          return pelaajat;
26      }
27  
28      public Tekoaly getTekoaly() {
29          return tekoaly;
30      }
31  
32      public void lisaaPelaaja(String nimi) {
33          this.pelaajat.add(new Pelaaja(nimi));
34      }
35  
36      public void jaaKortit() {
37          for (Pelaaja pelaaja : pelaajat) {
38              for (int i = 0; i < 5; i++) {
39                  try {
40                      pelaaja.lisaaKateen(sk.getPakka().otaEnsimmainenKortti());
41                  } catch (Exception e) {
42                      System.out.println(e.getMessage());
43                  }
44              }
45          }
46      }
47  
48      public void sekoitaPakka() {
49          sk.getPakka().sekoitaKortit();
50      }
51  
52      /**
53       * Metodi kertoo, että peli loppuu, jos vain yhdellä pelaajalla on kädessään
54       * kortteja.
55       *
56       * @return boolean
57       */
58      public boolean peliJatkuu() {
59          int pelissaMukana = 0;
60          for (Pelaaja pelaaja : pelaajat) {
61              if (!pelaaja.getKasi().onTyhja()) {
62                  pelissaMukana++;
63              }
64          }
65          return pelissaMukana != 1;
66      }
67  
68      public void asetaSeuraavaPelaajaVuoroon() {
69          for (int i = 0; i < pelaajat.size(); i++) {
70              if (pelaajat.get(i).equals(sk.nykyinenVuoro().getPelaaja())) {
71                  sk.nykyinenVuoro().setJatkuu(false);
72                  if (i == pelaajat.size() - 1) {
73                      sk.lisaaVuoro(pelaajat.get(0));
74                      break;
75                  } else {
76                      sk.lisaaVuoro(pelaajat.get(i + 1));
77                      break;
78                  }
79              }
80          }
81      }
82  }
