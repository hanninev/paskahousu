3   import joululabra.paskahousu.domain.Kortti;
4   import joululabra.paskahousu.domain.Korttijoukko;
5   import joululabra.paskahousu.domain.Pakka;
6   import joululabra.paskahousu.domain.Vuoro;

8   public class Saannot {
9   
10      /**
11       * Metodi kertoo, että pelaaja joutuu aina nostamaan pinon.
12       *
13       * Pelaaja joutuu aina nostamaan pinon, jos pinossa on tasan yksi kortti ja
14       * pinon ylimmän kortin arvo on 10 tai 14 (ässä).
15       *
16       * @param pino Nykyinen pino
17       *
18       * @return boolean
19       */
20      public static boolean pakkoNostaaPino(Korttijoukko pino) {
21          return ((pino.korttienMaara() == 1)
22                  && ((pino.viimeisinKortti().getArvo() == 10)
23                  || (pino.viimeisinKortti().getArvo() == 14)));
24      }
25  
26      /**
27       * Metodi kertoo, saako pelaaja kokeilla onneaan pakasta.
28       *
29       * Pelaaja saa kokeilla onneaan pakasta, jos ei ole saman vuoron aikana
30       * nostanut kortteja pakasta tai laittanut kortteja pinoon.
31       *
32       * @param vuoro Käsiteltävä vuoro
33       *
34       * @return boolean
35       */
36      public static boolean saaKokeillaOnnea(Vuoro vuoro) {
37          return ((vuoro.getNostetut().onTyhja()) && (vuoro.getLaitetut().onTyhja()));
38      }
39  
40      /**
41       * Metodi kertoo, onko vuorossa olevan pelaajan kädessä liian vähän
42       * kortteja.
43       *
44       * Kädessä on liian vähän kortteja, jos pakkaa on jäljellä ja kädessä on
45       * alle viisi korttia.
46       *
47       * @param vuoro Käsiteltävä vuoro
48       *
49       * @return boolean
50       */
51      public static boolean kadessaLiianVahanKortteja(Vuoro vuoro, Pakka pakka) {
52          return ((!pakka.onTyhja()) && (vuoro.getPelaaja().getKasi().korttienMaara() < 5));
53      }
54  
55      /**
56       * Metodi kertoo, jos pino kaatuu.
57       *
58       * Jos pinossa olevien korttien määrä on enemmän kuin yksi ja pinon
59       * viimeisimmän kortin arvo on 10 tai 14 (ässä), pino kaatuu.
60       *
61       * @param pino Nykyinen pino
62       * @param vuoro Käsiteltävä vuoro
63       *
64       * @return boolean
65       */
66      public static boolean pinoKaatuu(Korttijoukko pino, Vuoro vuoro) {
67          return ((pino.korttienMaara() > 1)
68                  && ((pino.viimeisinKortti().getArvo() == 10)
69                  || (pino.viimeisinKortti().getArvo() == 14)));
70      }
71  
72      /**
73       * Metodi kertoo, sopiiko kortti pinoon.
74       *
75       * Jos vuoron aikana laitettujen korttien joukko on tyhjä, sovelletaan
76       * vuoron ensimmäiseen siirtoon liittyviä sääntöjä. Muussa tapauksessa
77       * lisättävän kortin arvon tulee olla sama kuin edellisen kortin arvo.
78       *
79       * @param pino Nykyinen pino
80       * @param vuoro Käsiteltävä vuoro
81       * @param kortti Kokeiltava kortti
82       * @param pakka Nykyinen pakka
83       *
84       * @return boolean
85       */
86      public static boolean korttiSopii(Korttijoukko pino, Pakka pakka, Vuoro vuoro, Kortti kortti) {
87          if (vuoro.getLaitetut().onTyhja()) {
88              return vuoronEnsimmainenSiirtoOk(pakka, pino, kortti);
89          } else {
90              return vuoronToinenTaiUseampiSiirtoOk(vuoro, kortti);
91          }
92      }
93  
94      /**
95       * Metodi kertoo, sopiiko vuoron ensimmäisenä siirrettävä kortti pinoon.
96       *
97       * Vuoron ensimmäinen siirto sopii aina pinoon, jos lisättävän kortin arvo
98       * on 2 tai jos sekä pakka että pino ovat tyhjiä.
99       *
100      * Jos pakassa on kortteja, mutta pino on tyhjä, sopii tyhjään pinoon
101      * kortit, joiden arvo on pienempi kuin 11 (jätkä) tai tasan 14 (ässä).
102      *
103      * Jos sekä pakassa että pinossa on kortteja ja pinon viimeisin kortti ei
104      * ole arvoltaan 2, pinoon sopii kaikki kortit, joiden arvo on yhtä suuri
105      * tai suurempi kuin pinon ylin kortti.
106      *
107      * @param pino Nykyinen pino
108      * @param kortti Kokeiltava kortti
109      * @param pakka Nykyinen pakka
110      *
111      * @return boolean
112      */
113     public static boolean vuoronEnsimmainenSiirtoOk(Pakka pakka, Korttijoukko pino, Kortti kortti) {
114         if (kortti.getArvo() == 2) {
115             return true;
116         }
117 
118         if (pino.onTyhja()) {
119             if (pakka.onTyhja()) {
120                 return true;
121             } else if (kortti.getArvo() < 11 || kortti.getArvo() == 14) {
122                 return true;
123             }
124         } else {
125             if ((pino.viimeisinKortti().getArvo() != 2) && (pino.viimeisinKortti().getArvo() <= kortti.getArvo())) {
126                 return true;
127             }
128         }
129         return false;
130     }
131 
132     /**
133      * Metodi kertoo, sopiiko saman vuoron aikana laitettu seuraava kortti
134      * pakkaan, jos vuoron ensimmäinen kortti on laitettu.
135      *
136      * @param vuoro Käsiteltävä vuoro
137      * @param kortti Kokeiltava kortti
138      *
139      * @return boolean
140      */
141     public static boolean vuoronToinenTaiUseampiSiirtoOk(Vuoro vuoro, Kortti kortti) {
142         return ((vuoro.getLaitetut().getKortit().get(0).getArvo() == kortti.getArvo()) && (kortti.getArvo() != 2) && (kortti.getArvo() != 10) && (kortti.getArvo() != 14));
143     }
