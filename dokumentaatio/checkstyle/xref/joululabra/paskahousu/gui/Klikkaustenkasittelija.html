47      public void paivitaNakyma() {
48          buttonitJaKortit.clear();
49          kortitKadessa.removeAll();
50  
51          for (Kortti kortti : peli.getPelaajat().get(0).getKasi().getKortit()) {
52              ImageIcon imageIcon = haeKortinKuvaJaMuunnaKoko(kortti);
53              JButton korttiButton = new JButton(imageIcon);
54              korttiButton.setBackground(Color.WHITE);
55              korttiButton.addActionListener(this);
56              buttonitJaKortit.put(korttiButton, kortti);
57              kortitKadessa.add(korttiButton);
58          }
59  
60          kortitKadessa.setLayout(new GridLayout(1, buttonitJaKortit.size()));
61          kortitKadessa.validate();
62          kortitKadessa.repaint();
63  
64          vastustaja.setText("<html>" + peli.getPelaajat().get(1).getNimi() + ", kortteja kädessä " + peli.getPelaajat().get(1).getKasi().korttienMaara() + "<br><br>Tässä myös vastustajan kortit<br>testauksen helpottamiseksi:<br>" + peli.getPelaajat().get(1).getKasi().getKortit() + "</html>");
65          pino.setText("Pinossa " + peli.getSk().getPino().korttienMaara() + " korttia.");
66          pino.setIcon(haeKortinKuvaJaMuunnaKoko(peli.getSk().getPino().viimeisinKortti()));
67          pino.setHorizontalTextPosition(JLabel.CENTER);
68          pino.setVerticalTextPosition(JLabel.BOTTOM);
69          pakka.setText("Kortteja pakassa: " + peli.getSk().getPakka().korttienMaara());
70  
71          pakka.repaint();
72      }

74      public ImageIcon haeKortinKuvaJaMuunnaKoko(Kortti kortti) {
75          if (kortti == null) {
76              kortti = new Kortti(Kortti.HERTTA, 0);
77          }
78          ImageIcon imageIcon = new ImageIcon(Kortti.MAAT[kortti.getMaa()].toLowerCase() + kortti.getArvo() + ".png");
79          Image image = imageIcon.getImage();
80          Image newimg = image.getScaledInstance(100, 145, java.awt.Image.SCALE_SMOOTH);
81          imageIcon = new ImageIcon(newimg);
82          return imageIcon;
83      }

85      @Override
86      public void actionPerformed(ActionEvent ae) {
87          try {
88              sb = new StringBuilder();
89  
90              for (JButton jbutton : buttonitJaKortit.keySet()) {
91                  if (ae.getSource() == (JButton) jbutton) {
92                      peli.getSk().siirraKorttiPinoon(buttonitJaKortit.get(jbutton));
93                      valmis.setEnabled(true);
94                      kokeileOnnea.setEnabled(false);
95                      nostaPino.setEnabled(false);
96                  }
97              }
98  
99              if (ae.getSource() == kokeileOnnea) {
100                 peli.getSk().kokeileOnnea();
101                 if (peli.getSk().nykyinenVuoro().getNostetut().getKortit().size() > 1) {
102                     sb.append("Sait pakasta kortin " + peli.getSk().nykyinenVuoro().getNostetut().getKortit().get(0) + ". Kortti ei käynyt pinoon, joten jouduit nostamaan pinon.<br>");
103                 } else {
104                     sb.append("Sait pakasta kortin " + peli.getSk().nykyinenVuoro().getNostetut().getKortit().get(0) + ". Sinulla kävi onni. Kortti oli sopiva pinoon.<br>");
105                 }
106                 kasitteleVuoro();
107             } else if (ae.getSource() == nostaPino) {
108                 peli.getSk().nostaPino();
109                 kasitteleVuoro();
110             } else if (ae.getSource() == valmis) {
111                 peli.getSk().taydennaKasi();
112                 peli.asetaSeuraavaPelaajaVuoroon();
113                 kasitteleVuoro();
114             }
115 
116             paivitaNakyma();
117 
118         } catch (Exception e) {
119             viestikentta.setText("<html><br><br><br><center>" + e.getMessage() + "</center></html>");
120         }
121     }

130     public void kasitteleVuoro() throws Exception {
131         if (!peli.peliJatkuu()) {
132             sb.append(peli.getSk().nykyinenVuoro().getPelaaja().getNimi().toString() + " hävisi!<br>");
133             valmis.setEnabled(false);
134             kokeileOnnea.setEnabled(false);
135             nostaPino.setEnabled(false);
136             kortitKadessa.setEnabled(false);
137         } else {
138 
139             if (Saannot.pakkoNostaaPino(peli.getSk().getPino())) {
140                 if (peli.getSk().nykyinenVuoro().getPelaaja().equals(peli.getPelaajat().get(0))) {
141                     sb.append("Sinä jouduit nostamaan kortin.<br>");
142                 } else {
143                     sb.append(peli.getSk().nykyinenVuoro().getPelaaja().getNimi() + " joutui nostamaan kortin.<br>");
144                 }
145                 peli.getSk().nostaPino();
146                 kortitKadessa.setEnabled(false);
147                 kokeileOnnea.setEnabled(false);
148                 nostaPino.setEnabled(true);
149             }
150 
151             valmis.setEnabled(false);
152             kokeileOnnea.setEnabled(true);
153             nostaPino.setEnabled(true);
154 
155             if ((peli.getSk().getPakka().onTyhja()) || (!Saannot.saaKokeillaOnnea(peli.getSk().nykyinenVuoro()))) {
156                 kokeileOnnea.setEnabled(false);
157             }
158             if (peli.getSk().getPino().onTyhja()) {
159                 nostaPino.setEnabled(false);
160             }
161 
162             paivitaNakyma();
163 
164             if (!peli.getSk().nykyinenVuoro().getPelaaja().isTekoaly()) {
165                 return;
166             }
167             vastustajaSiirtaa();
168         }
169     }

171     public void vastustajaSiirtaa() throws Exception {
172         Integer pinonKoko = peli.getSk().getPino().korttienMaara();
173         peli.getTekoaly().valitseToiminto();
174         if ((peli.getSk().nykyinenVuoro().getNostetut().korttienMaara() == pinonKoko) && (pinonKoko != 0)) {
175             sb.append("Vastustaja nosti pinon. ");
176         } else if (peli.getSk().nykyinenVuoro().getNostetut().korttienMaara() == 1) {
177             sb.append("Vastustaja kokeili onneaan ja kortti oli sopiva. <br>");
178         } else if (peli.getSk().nykyinenVuoro().getNostetut().korttienMaara() == 1 + pinonKoko) {
179             sb.append("Vastustaja kokeili onneaan ja joutui nostamaan pinon. <br>");
180         }
181         if (!peli.getSk().nykyinenVuoro().getLaitetut().onTyhja()) {
182             sb.append("Vastustaja laittoi pinoon kortit: " + peli.getSk().nykyinenVuoro().getLaitetut().getKortit() + " <br>");
183         }
184         sb.append("Sinun vuorosi.");
185         viestikentta.setText("<html><br><center>" + sb.toString() + "</center></html>");
186         peli.getSk().taydennaKasi();
187         peli.asetaSeuraavaPelaajaVuoroon();
188         kasitteleVuoro();
189     }
