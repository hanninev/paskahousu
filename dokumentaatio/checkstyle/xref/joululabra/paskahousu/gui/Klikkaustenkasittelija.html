35      public Klikkaustenkasittelija(Peli peli, JLabel vastustaja, JLabel pino, JLabel pakka, JPanel kortitKadessa, JButton kokeileOnnea, JButton nostaPino, JButton valmis, JLabel viestikentta) {

37          this.kokeileOnnea = kokeileOnnea;

49      public void paivitaNakyma() {
50          painikkeetJaKortit.clear();
51          kortitKadessa.removeAll();
52  
53          for (Kortti kortti : peli.getPelaajat().get(0).getKasi().getKortit()) {
54              ImageIcon imageIcon = haeKortinKuvaJaMuunnaKoko(kortti);
55              JButton korttiPainike = new JButton(imageIcon);
56              korttiPainike.setHorizontalAlignment(SwingConstants.LEFT);
57              korttiPainike.setBackground(Color.WHITE);
58              korttiPainike.setBorder(BorderFactory.createEmptyBorder(0, 0, 0, 0));
59              korttiPainike.addActionListener(this);
60              painikkeetJaKortit.put(korttiPainike, kortti);
61              kortitKadessa.add(korttiPainike);
62          }
63  
64          kortitKadessa.setLayout(new GridLayout(1, painikkeetJaKortit.size()));
65          kortitKadessa.validate();
66          kortitKadessa.repaint();
67  
68          vastustaja.setText("<html>" + peli.getPelaajat().get(1).getNimi() + ", kortteja kädessä " + peli.getPelaajat().get(1).getKasi().korttienMaara());
69          pino.setText("Pinossa " + peli.getSk().getPino().korttienMaara() + " korttia.");
70          pino.setIcon(haeKortinKuvaJaMuunnaKoko(peli.getSk().getPino().viimeisinKortti()));
71          pino.setHorizontalTextPosition(JLabel.CENTER);
72          pino.setVerticalTextPosition(JLabel.BOTTOM);
73          pakka.setText("Kortteja pakassa: " + peli.getSk().getPakka().korttienMaara());
74  
75          pakka.repaint();
76      }

78      public ImageIcon haeKortinKuvaJaMuunnaKoko(Kortti kortti) {
79          if (kortti == null) {
80              kortti = new Kortti(Kortti.HERTTA, 0);
81          }
82          ImageIcon imageIcon = new ImageIcon(Klikkaustenkasittelija.class.getResource("images/" + Kortti.MAAT[kortti.getMaa()].toLowerCase() + kortti.getArvo() + ".png"));
83          Image image = imageIcon.getImage();
84          Image newimg = image.getScaledInstance(100, 145, java.awt.Image.SCALE_SMOOTH);
85          imageIcon = new ImageIcon(newimg);
86          return imageIcon;
87      }

89      @Override
90      public void actionPerformed(ActionEvent ae) {
91          try {
92              sb = new StringBuilder();
93  
94              for (JButton jbutton : painikkeetJaKortit.keySet()) {
95                  if (ae.getSource() == (JButton) jbutton) {
96                      peli.getSk().siirraKorttiPinoon(painikkeetJaKortit.get(jbutton));
97                      valmis.setEnabled(true);
98                      kokeileOnnea.setEnabled(false);
99                      nostaPino.setEnabled(false);
100                 }
101             }
102 
103             if (ae.getSource() == kokeileOnnea) {
104                 peli.getSk().kokeileOnnea();
105                 kokeileOnnea.setEnabled(false);
106                 valmis.setEnabled(false);
107                 sb.append("Sait pakasta kortin " + peli.getSk().nykyinenVuoro().getPelaaja().getKasi().viimeisinKortti().toString() + ". <br>Jos mikään kortti ei käy pinoon, sinun on pakko nostaa pino.");
108             } else if (ae.getSource() == nostaPino) {
109                 peli.getSk().nostaPino();
110                 sb.append("Nostit pinon. Vuoro siirtyi vastustajalle.<br>");
111                 kasittelePainikkeet();
112             } else if (ae.getSource() == valmis) {
113                 peli.getSk().kaadaPinoJosKaatuuJaTaydennaKasi();
114                 if (peli.getSk().nykyinenVuoro().isKaatoiPinon()) {
115                     sb.append("Kaadoit pinon. Saat uuden vuoron.<br>");
116                 }
117                 kasittelePainikkeet();
118             }
119 
120             paivitaViestikentta();
121             paivitaNakyma();
122 
123         } catch (Exception e) {
124             sb.append(e.getMessage().toString());
125             paivitaViestikentta();
126         }
127     }

133     public void kasittelePainikkeet() throws Exception {
134         if (!peli.peliJatkuu()) {
135             ilmoitaPelinPaattyminen();
136             return;
137         } else {
138             peli.getSk().vuoronVaihtuminen();
139             paivitaViestikentta();
140         }
141 
142         valmis.setEnabled(false);
143         kokeileOnnea.setEnabled(true);
144         nostaPino.setEnabled(true);
145 
146         if ((peli.getSk().getPakka().onTyhja()) || (!Saannot.saaKokeillaOnnea(peli.getSk().nykyinenVuoro()))) {
147             kokeileOnnea.setEnabled(false);
148         }
149         if (peli.getSk().getPino().onTyhja()) {
150             nostaPino.setEnabled(false);
151         }
152         if ((peli.getSk().getPino().onTyhja()) && (!peli.getSk().getPakka().onTyhja()) && (peli.getSk().nykyinenVuoro().getPelaaja().pieninKortti().getArvo() >= 11) && (peli.getSk().nykyinenVuoro().getPelaaja().suurinKortti().getArvo() < 14)) {
153             toiminnanTeksti("Pakkaa on jäljellä, joten mikään kädessä olevista korteistasi ei sovi tyhjään pöytään.<br>Voit luopua vuorostasi tai kokeilla onneasi pakasta, jos et jo tehnyt sitä.<br>", "");
154             valmis.setEnabled(true);
155         }
156 
157         paivitaNakyma();
158 
159         if (!peli.getSk().nykyinenVuoro().getPelaaja().isTekoaly()) {
160             sb.append("Sinun vuorosi.<br>");
161             paivitaViestikentta();
162             return;
163         }
164         vastustajaSiirtaa();
165     }

167     private void ilmoitaPelinPaattyminen() {
168         toiminnanTeksti("Sinä voitit, onnea!", "Vastustaja voitti!");
169         valmis.setEnabled(false);
170         kokeileOnnea.setEnabled(false);
171         nostaPino.setEnabled(false);
172     }

174     public void vastustajaSiirtaa() throws Exception {
175         Integer pinonKoko = peli.getSk().getPino().korttienMaara();
176         Korttijoukko lisatyt = peli.getTekoaly().valitseToiminto();
177         peli.getSk().kaadaPinoJosKaatuuJaTaydennaKasi();
178         if (peli.getSk().nykyinenVuoro().getLaitettuPinoon() != 0) {
179             sb.append("Vastustaja laittoi pinoon " + lisatyt.toString() + " <br>");
180         }
181         tekstitVastustajanTekemisesta(pinonKoko);
182 
183         paivitaViestikentta();
184         kasittelePainikkeet();
185     }
186 
187     private void tekstitVastustajanTekemisesta(Integer pinonKoko) {
188 
189         if (peli.getSk().nykyinenVuoro().isKokeiliOnnea()) {
190             sb.append("Vastustaja kokeili onneaan, ja hän ");
191             if (peli.getSk().getPino().korttienMaara() >= pinonKoko + 1) {
192                 sb.append("laittoi pinoon sopivan kortin.<br>");
193             } else {
194                 sb.append("joutui nostamaan pinon.<br>");
195             }
196         } else if (peli.getSk().nykyinenVuoro().isNostiPinon()) {
197             sb.append("Vastustaja nosti pinon. <br>");
198         }
199 
200         if (peli.getSk().nykyinenVuoro().isKaatoiPinon()) {
201             sb.append("Vastustaja kaatoi pinon ja sai uuden vuoron.<br>");
202         }
203     }

205     private void toiminnanTeksti(String kayttaja, String vastustaja) {
206         if (peli.getSk().nykyinenVuoro().getPelaaja().equals(peli.getPelaajat().get(0))) {
207             sb.append(kayttaja);
208         } else if (peli.getSk().nykyinenVuoro().getPelaaja().equals(peli.getPelaajat().get(1))) {
209             sb.append(vastustaja);
210         }
211     }
